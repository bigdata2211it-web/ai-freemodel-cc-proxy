Управление пулом ключей

# список ключей (маскированы, со статусом и счётчиками)
GET /api/keys
→ { "total":2, "current":0, "keys":[{ "index":0, "masked":"fe_oa_1e…a242", "current":true, "ok":3, "bad":false, "count":3, "lastStatus":200 }] }

# добавить один или несколько (персистится, дубликаты игнорируются)
POST /api/keys        { "keys":["fe_oa_…","fe_oa_…"] }

# удалить по индексу или по точному ключу
DELETE /api/keys     { "index":0 }   # или { "key":"fe_oa_…" }

# проб по порядку, залочиться на первом рабочем
POST /api/keys/find   {}

Клиенты — подключение

# Factory Droid (~/.factory/settings.json customModels, + FACTORY_AIRGAP_ENABLED=1)
{ "model":"claude-opus-4-8", "baseUrl":"http://127.0.0.1:11440", "apiKey":"dummy",
  "provider":"anthropic", "maxOutputTokens":16384 }

# Claude Code
export ANTHROPIC_BASE_URL=http://127.0.0.1:11440
export ANTHROPIC_API_KEY=dummy
claude "…"

# Pi CLI (~/.pi/agent/models.json)
{ "providers":{ "freemodel":{ "baseUrl":"http://127.0.0.1:11440",
  "api":"anthropic-messages", "apiKey":"dummy",
  "models":[{ "id":"claude-opus-4-8", "reasoning":true, "input":["text","image"] }] } } }

# OpenAI SDK / Cursor / Cline (OpenAI-протокол, любая модель)
base_url=http://127.0.0.1:11440/v1  api_key=dummy  model=gpt-5.4-mini # или claude-opus-4-8

CLI ключей

node keys.js add fe_oa_…      # добавить
node keys.js                   # список
node keys.js rm 0              # удалить по индексу
node keys.js status            # сводка
