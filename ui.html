Manage the key pool

# list keys (masked, with status + counts)
GET /api/keys
→ { "total":2, "current":0, "keys":[{ "index":0, "masked":"fe_oa_1e…a242", "current":true, "ok":3, "limited":0, "bad":false, "count":3, "lastStatus":200, "lastUsed":"…" }] }

# add one or many keys (persisted, dedup'd)
POST /api/keys        { "keys":["fe_oa_…","fe_oa_…"] }
→ { "ok":true, "added":2, "total":4 }

# remove by index (from GET) or by exact key
DELETE /api/keys     { "index":0 }     # or  { "key":"fe_oa_…" }

# probe forward, lock onto the FIRST working key (minimal requests)
POST /api/keys/find   {}
→ { "ok":true, "current":1, "masked":"fe_oa_9b…1d39", "probed":2, "tried":[…] }
