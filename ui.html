Базовый URL http://127.0.0.1:11440 · авторизация необязательна (любое значение; прокси подставляет реальный ключ из пула).

Управление пулом ключей

# список ключей (маскированы, со статусом и счётчиками)
GET /api/keys
→ { "total":2, "current":0, "keys":[{ "index":0, "masked":"fe_oa_1e…a242", "current":true, "ok":3, "limited":0, "bad":false, "count":3, "lastStatus":200, "lastUsed":"…" }] }

# добавить один или несколько (персистится, дубликаты игнорируются)
POST /api/keys        { "keys":["fe_oa_…","fe_oa_…"] }
→ { "ok":true, "added":2, "total":4 }

# удалить по индексу (из GET) или по точному ключу
DELETE /api/keys     { "index":0 }     # или  { "key":"fe_oa_…" }

# проб по порядку, залочиться на ПЕРВОМ рабочем (минимум запросов)
POST /api/keys/find   {}
→ { "ok":true, "current":1, "masked":"fe_oa_9b…1d39", "probed":2, "tried":[…] }

curl — добавить ключ из скрипта

curl -X POST http://127.0.0.1:11440/api/keys \
  -H 'content-type: application/json' -d '{"keys":["fe_oa_…"]}'
# или встроенная CLI:  node keys.js add fe_oa_…
